'''Multiply Command'''
# pylint: disable=missing-class-docstring
from app.command_handler import Command, CommandHandler

class MenuCommand(Command):
    def execute(self):
        '''Displays available plugins'''
        print("Available plugins:")
        for index, (plugin_name, plugin_class) in enumerate(app.loaded_plugins.items(), start=1):
            print(f"{index}. {plugin_name}")
